/**
 * Cek apakah user adalah pelajar Ma'had
 */
export function isMahadUser(profile) {
  return ['idadi', 'tsanawi_adabi', 'tsanawi_ilmi'].includes(profile?.level);
}

/**
 * Ambil config level Ma'had berdasarkan ID
 */
export function getMahadLevel(levelId) {
  return MAHAD_LEVELS[levelId] || null;
}

/**
 * Filter Maddah berdasarkan jenjang user
 */
export function getMahadMaddahByLevel(levelId) {
  return MAHAD_MADDAH.filter((m) => m.jenjang.includes(levelId));
}

/**
 * Generate starter pack text untuk user Ma'had
 */
export function generateMahadStarterPack(profile) {
  const level = getMahadLevel(profile.level);
  if (!level) return '';

  const ageRange = profile.level === 'idadi' ? '12-15' : '15-18';
  const jurusanText =
    profile.level === 'tsanawi_ilmi'
      ? 'Ilmi (علمي) — Sains & Matematika'
      : profile.level === 'tsanawi_adabi'
      ? 'Adabi (أدبي) — Sastra & Sosial'
      : null;

  const maddahUmum =
    profile.level === 'tsanawi_ilmi'
      ? 'Matematika, Fisika, Kimia, Biologi, Bahasa Inggris'
      : profile.level === 'tsanawi_adabi'
      ? 'Matematika, Sejarah, Geografi, Filsafat & Mantiq, Statistik, Bahasa Inggris'
      : 'Matematika, Sains (IPA), Bahasa Inggris, Studi Sosial';

  const challengeLabels = (profile.challenges || [])
    .map((id) => MAHAD_CHALLENGES.find((c) => c.id === id)?.label)
    .filter(Boolean)
    .map((l) => `• ${l}`)
    .join('\n');

  return `Halo! Sebelum kita mulai belajar bareng, aku mau kenalin diri dulu ya.

══════════════════════════════════
👤 SIAPA AKU
══════════════════════════════════
Aku pelajar Indonesia yang sekolah di Ma'had Al-Azhar Mesir.
Tingkat : ${level.label} Tahun ${profile.year || '[isi tahunmu]'}
${jurusanText ? `Jurusan : ${jurusanText}` : ''}
Usia    : sekitar ${ageRange} tahun

══════════════════════════════════
📚 MATA PELAJARAN YANG AKU PELAJARI
══════════════════════════════════
Agama  : Al-Qur'an, Tauhid, Fiqh, Nahwu, Sharaf, Hadits, Sirah
Umum   : ${maddahUmum}

/**
 * Simpan profil Ma'had ke localStorage
 */
export function saveMahadProfile(profileData) {
  const existing = JSON.parse(localStorage.getItem('userProfile') || '{}');
  const updated = {
    ...existing,
    ...profileData,
    isMahad: true,
    updatedAt: new Date().toISOString(),
  };
  localStorage.setItem('userProfile', JSON.stringify(updated));
  return updated;
}

/**
 * Ambil profil user dari localStorage
 */
export function getUserProfile() {
  return JSON.parse(localStorage.getItem('userProfile') || '{}');
}

/**
 * Ambil rekomendasi Maddah berdasarkan tantangan user
 */
export function getRecommendedMaddah(profile) {
  if (!isMahadUser(profile)) return [];

  const byLevel = getMahadMaddahByLevel(profile.level);
  const challenges = profile.challenges || [];

  // Map challenge ke Maddah yang relevan
  const challengeMaddahMap = {
    math_arabic: ['riyadhiyat'],
    science_arabic: ['fisika', 'kimia', 'biologi'],
    social_arabic: ['tarikh', 'jughrafiya'],
    quran_hafalan: ['quran-tajwid'],
    insya: ['insya-tabir'],
    fahm_maqru: ['insya-tabir', 'nahwu-sharaf-dasar'],
    nahwu_sharaf: ['nahwu-sharaf-dasar'],
    english: ['bahasa-inggris'],
  };

  const recommendedIds = new Set();
  challenges.forEach((challenge) => {
    const ids = challengeMaddahMap[challenge] || [];
    ids.forEach((id) => recommendedIds.add(id));
  });

  // Prioritaskan yang direkomendasi, sisanya setelahnya
  const recommended = byLevel.filter((m) => recommendedIds.has(m.id));
  const others = byLevel.filter((m) => !recommendedIds.has(m.id));

  return [...recommended, ...others];
}

import { useState } from 'react';
import { MAHAD_LEVELS, MAHAD_CHALLENGES, saveMahadProfile, generateMahadStarterPack } from '../utils/mahad-helpers';

export default function MahadOnboarding({ level, onComplete }) {
  const [step, setStep] = useState(1); // 1: tahun, 2: tantangan, 3: gaya belajar
  const [selectedYear, setSelectedYear] = useState(null);
  const [selectedChallenges, setSelectedChallenges] = useState([]);
  const [learningStyle, setLearningStyle] = useState(null);

  const levelConfig = MAHAD_LEVELS[level];

  const LEARNING_STYLES = [
    { id: 'visual', label: 'Visual', desc: 'Lebih mudah dengan diagram, tabel, mind map', icon: '👁️' },
    { id: 'auditory', label: 'Auditori', desc: 'Lebih mudah kalau dengar penjelasan', icon: '👂' },
    { id: 'discussion', label: 'Diskusi', desc: 'Lebih mudah kalau ada teman belajar', icon: '💬' },
    { id: 'memorizer', label: 'Penghafal', desc: 'Kuat di hafalan dan pengulangan', icon: '🧠' },
    { id: 'analytical', label: 'Analitis', desc: 'Suka bongkar konsep sampai ke dasarnya', icon: '🔍' },
  ];

  function toggleChallenge(id) {
    setSelectedChallenges((prev) =>
      prev.includes(id) ? prev.filter((c) => c !== id) : [...prev, id]
    );
  }

  function handleComplete() {
    const profile = {
      level,
      year: selectedYear,
      challenges: selectedChallenges,
      learningStyle,
      isMahad: true,
    };
    saveMahadProfile(profile);
    const starterPack = generateMahadStarterPack(profile);
    onComplete({ profile, starterPack });
  }

  return (
    <div className="mahad-onboarding">
      {/* Progress bar */}
      <div className="onboarding-progress">
        <div
          className="onboarding-progress-fill"
          style={{ width: `${(step / 3) * 100}%` }}
        />
      </div>
      <p className="onboarding-step-label">LANGKAH {step} DARI 3</p>

      {/* Level badge */}
      <div
        className="mahad-level-badge"
        style={{ backgroundColor: levelConfig.badgeColor + '20', borderColor: levelConfig.badgeColor }}
      >
        <span>{levelConfig.icon}</span>
        <span style={{ color: levelConfig.badgeColor }}>{levelConfig.label}</span>
        <span className="badge-arabic">{levelConfig.arabicLabel}</span>
      </div>

      {/* Step 1: Tahun */}
      {step === 1 && (
        <div className="onboarding-step">
          <h2 className="onboarding-question">Kamu di tahun ke berapa?</h2>
          <p className="onboarding-sub">Di {levelConfig.label} {levelConfig.subtitle}</p>
          <div className="year-options">
            {levelConfig.years.map((year) => (
              <button
                key={year}
                className={`year-option ${selectedYear === year ? 'selected' : ''}`}
                onClick={() => setSelectedYear(year)}
              >
                {year}
              </button>
            ))}
          </div>
          <button
            className="btn-next"
            disabled={!selectedYear}
            onClick={() => setStep(2)}
          >
            Lanjut →
          </button>
        </div>
      )}

      {/* Step 2: Tantangan */}
      {step === 2 && (
        <div className="onboarding-step">
          <h2 className="onboarding-question">Apa tantangan belajarmu?</h2>
          <p className="onboarding-sub">Pilih semua yang terasa sulit (boleh lebih dari satu)</p>
          <div className="challenge-options">
            {MAHAD_CHALLENGES.map((challenge) => (
              <button
                key={challenge.id}
                className={`challenge-option ${selectedChallenges.includes(challenge.id) ? 'selected' : ''}`}
                onClick={() => toggleChallenge(challenge.id)}
              >
                <span className="challenge-icon">{challenge.icon}</span>
                <span>{challenge.label}</span>
                {selectedChallenges.includes(challenge.id) && (
                  <span className="challenge-check">✓</span>
                )}
              </button>
            ))}
          </div>
          <div className="step-nav">
            <button className="btn-back" onClick={() => setStep(1)}>← Kembali</button>
            <button
              className="btn-next"
              disabled={selectedChallenges.length === 0}
              onClick={() => setStep(3)}
            >
              Lanjut →
            </button>
          </div>
        </div>
      )}

      {/* Step 3: Gaya belajar */}
      {step === 3 && (
        <div className="onboarding-step">
          <h2 className="onboarding-question">Gimana kamu paling enak belajar?</h2>
          <p className="onboarding-sub">Pilih satu yang paling cocok</p>
          <div className="style-options">
            {LEARNING_STYLES.map((style) => (
              <button
                key={style.id}
                className={`style-option ${learningStyle === style.id ? 'selected' : ''}`}
                onClick={() => setLearningStyle(style.id)}
              >
                <span className="style-icon">{style.icon}</span>
                <div>
                  <p className="style-label">{style.label}</p>
                  <p className="style-desc">{style.desc}</p>
                </div>
              </button>
            ))}
          </div>
          <div className="step-nav">
            <button className="btn-back" onClick={() => setStep(2)}>← Kembali</button>
            <button
              className="btn-next btn-complete"
              disabled={!learningStyle}
              onClick={handleComplete}
            >
              Lihat Starter Pack-ku →
            </button>
          </div>
        </div>
      )}
    </div>
  );
}

import { useState } from 'react';

export default function MahadStarterPack({ starterPack, onContinue }) {
  const [copied, setCopied] = useState(false);
  const [editedPack, setEditedPack] = useState(starterPack);

  function handleCopy() {
    navigator.clipboard.writeText(editedPack);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  }

  return (
    <div className="starter-pack-screen">
      {/* Header */}
      <div className="sp-header">
        <p className="sp-arabic">أَهْلاً وَسَهْلاً</p>
        <h2 className="sp-title">Starter Pack-mu Siap! 🎉</h2>
        <p className="sp-subtitle">
          Paste teks ini ke AI favoritmu sebelum mulai belajar.
          AI akan langsung paham siapa kamu dan cara terbaik membantumu.
        </p>
      </div>

      {/* Starter pack text box */}
      <div className="sp-textbox-wrapper">
        <textarea
          className="sp-textbox"
          value={editedPack}
          onChange={(e) => setEditedPack(e.target.value)}
          rows={20}
        />
      </div>

      {/* Actions */}
      <div className="sp-actions">
        <button className="sp-btn-copy" onClick={handleCopy}>

import { useState } from 'react';
import { MAHAD_MADDAH, getMahadMaddahByLevel } from '../utils/mahad-helpers';

const FILTERS = [
  { id: 'all', label: 'Semua' },
  { id: 'agama', label: 'Maddah Agama' },
  { id: 'umum-semua', label: 'Maddah Umum' },
  { id: 'umum-ilmi', label: 'Tsanawi Ilmi' },
  { id: 'umum-adabi', label: 'Tsanawi Adabi' },
];

export default function MahadMaddahGrid({ userLevel }) {
  const [activeFilter, setActiveFilter] = useState('all');

  const allMaddah = getMahadMaddahByLevel(userLevel);
  const filtered =
    activeFilter === 'all'
      ? allMaddah
      : allMaddah.filter((m) => m.category === activeFilter);

  return (
    <div className="mahad-maddah-section">
      {/* Filter chips */}
      <div className="mahad-filter-bar">
        {FILTERS.map((f) => (
          <button
            key={f.id}
            className={`mahad-filter-chip ${activeFilter === f.id ? 'active' : ''}`}
            onClick={() => setActiveFilter(f.id)}
          >
            {f.label}
          </button>
        ))}
      </div>

      {/* Maddah grid */}
      <div className="mahad-maddah-grid">
        {filtered.map((maddah) => (
          <MahadMaddahCard key={maddah.id} maddah={maddah} />
        ))}
      </div>

      {filtered.length === 0 && (
        <div className="mahad-empty">
          <p>Tidak ada Maddah untuk filter ini.</p>
        </div>
      )}
    </div>
  );
}

import { useState } from 'react';
import { MAHAD_PROMPTS } from '../data/mahad-data';

export default function MahadPromptLibrary() {
  const [activeSubCat, setActiveSubCat] = useState('all');

  const subCategories = ['all', ...new Set(MAHAD_PROMPTS.map((p) => p.subCategory))];

  const filtered =
    activeSubCat === 'all'
      ? MAHAD_PROMPTS
      : MAHAD_PROMPTS.filter((p) => p.subCategory === activeSubCat);

  return (
    <div className="mahad-prompt-library">
      {/* Sub-category filter */}
      <div className="prompt-subcat-filter">
        {subCategories.map((cat) => (
          <button
            key={cat}
            className={`prompt-subcat-chip ${activeSubCat === cat ? 'active' : ''}`}
            onClick={() => setActiveSubCat(cat)}
          >
            {cat === 'all' ? 'Semua' : cat}
          </button>
        ))}
      </div>

      {/* Prompt grid */}
      <div className="mahad-prompt-grid">
        {filtered.map((prompt) => (
          <MahadPromptCard key={prompt.id} prompt={prompt} />
        ))}
      </div>
    </div>
  );
}

function MahadPromptCard({ prompt }) {
  const [copied, setCopied] = useState(false);
  const [expanded, setExpanded] = useState(false);

  function handleCopy() {
    navigator.clipboard.writeText(prompt.template);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  }

  return (
    <div className="mahad-prompt-card">
      {/* Badge */}
      <div className="mahad-prompt-badge">MA'HAD</div>

      {/* Sub-category label */}
      <p className="mahad-prompt-subcat">{prompt.subCategory}</p>

      {/* Title */}
      <h3 className="mahad-prompt-title">{prompt.title}</h3>

      {/* Situation */}
      <p className="mahad-prompt-situation">📌 {prompt.situation}</p>

      {/* Template preview / full */}
      <div className={`mahad-prompt-template ${expanded ? 'expanded' : 'collapsed'}`}>
        <pre>{prompt.template}</pre>
      </div>

      {/* Actions */}
      <div className="mahad-prompt-actions">
        <button
          className="mahad-prompt-expand"
          onClick={() => setExpanded(!expanded)}
        >
          {expanded ? 'Sembunyikan ▲' : 'Lihat Prompt ▼'}
        </button>
        <button
          className={`mahad-prompt-copy ${copied ? 'copied' : ''}`}
          onClick={handleCopy}
        >
          {copied ? '✓ Tersalin!' : '📋 Salin'}
        </button>
      </div>
    </div>
  );
}
